[tex] \cos \: x = \frac{a}{1} [/tex]

[tex] \cos \: x = \sqrt{1 {}^{2} - a {}^{2} } [/tex]

[tex] \cos \: x = \sqrt{1 - a {}^{2} } [/tex]

[tex] \cot \: x = \frac{a}{ \sqrt{1 - a {}^{2} } } [/tex]

♤Lalu,tentukan nilai cot y:

[tex] \sin \: y = \frac{b}{1} [/tex]

[tex] \sin \: y = \sqrt{1 {}^{2} - b {}^{2} } [/tex]

[tex] \sin \: y = \sqrt{1 - b {}^{2} } [/tex]

[tex] \cot \: y = \frac{ \sqrt{1 - b {}^{2} } }{b} [/tex]

♧Maka,nilai cot x - cot y yaitu:

[tex] = \frac{a}{ \sqrt{1 - a {}^{2} } } - \frac{ \sqrt{1 - b {}^{2} } }{b} [/tex]
