titik P di tengah-tengah AD

[tex]s_1=s_2[/tex]

[tex]v_1t_1=v_2t_2[/tex]

[tex]c\cdot60=0,8c\cdot t[/tex]

[tex]\frac{60}{0,8}=t[/tex]

[tex]\frac{60\cdot5}{4}=t[/tex]

[tex]75=t[/tex]
